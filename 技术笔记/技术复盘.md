- 数组： 
  - 数据结构的一种，也是语言自带的数据结构
  - 标识：[] 
  - 定义格式： 
    - 数据类型[] 数组变量名=new 数据类型[长度];
    - 数据类型[] 数组变量名={元素1,元素2,...}; 直接初始化
  - 特性：
    - 在内存中是一段连续的空间 
    - 可以根据第一个元素的地址，推断后面元素的地址
    - 第一个元素的地址，会转为下标0，后面的元素的位置依次递增1
    - 因为是连续的空间，定义之后，数组后面的空间有可能被其他变量的占用
    - 数组在定义之后不支持修改长度 
  - 操作： 
    - 数组名[下标]=值;
    - 数据类型 变量名= 数组名[下标];
  - 样例： 
    - int[] arr = new int[10];
    - int[] arr1 = {1,2,3,4,5,6,71,1,2,2,3,4};
    - arr[1]=100;// 向arr数组中第二个位置存入100
    - int n = arr1[6];

```java
package com;

import java.util.Scanner;

public class ArrDemo {
    public static void main(String[] args) {

        // 1: 输入一个数字n 表示第二行会输入n个整数
        // 2： 输出这n个数之和
        // 3： 输出 这n个数中最大值 最小值 平均值

        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int[] arr = new int[n];// 可以存储n个数
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }
        int sum = 0;
        int max = arr[0];
        int min = arr[0];
        for (int i = 0; i < n; i++) {
            int num = arr[i];
            sum = sum + num;
            if (num > max) {
                max = num;
            }
            if (num < min) {
                min = num;
            }
        }
        System.out.println("总和：" + sum);
        System.out.println(min);
        System.out.println(max);
        System.out.println(sum / n);
        // 4: 逆序输出所有元素
        for (int i = n-1; i >=0 ; i--) {
            System.out.print(arr[i]+" ");
        }
        System.out.println();
        // 5： 判断每个元素的前面有几个小于自己的
        // 1 2 3 4 5
        for (int i = 0; i < n; i++) {
            int count=0;
            for (int j = 0; j < i; j++) {
                if(arr[j]<arr[i]){
                    count++;
                }
            }
            System.out.print(count+" ");
        }
    }
}
```
